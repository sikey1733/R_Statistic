#     ()
imt_index <- Obesity %>%
  filter(Age > 19) %>% 
  mutate(
    IMT_index = Weight/(Height^2)
  )

#   ,     
transformation <- imt_index %>%
  select(IMT_index, FAVC, CAEC, SMOKE, SCC, CALC, MTRANS,
         family_history_with_overweight, Age,
         Weight, Height, FCVC, NCP, CH2O, FAF, TUE) %>%
  mutate(
    across(c(FAVC, CAEC, SMOKE, SCC, CALC, MTRANS, family_history_with_overweight), ~ case_when(
      . == "yes" ~ 1,
      . == "no" ~ 0,
      . == "Always" ~ 1,
      . == "Frequently" ~ 2,
      . == "Sometimes" ~ 3,
      . == "Automobile" ~ 1,
      . == "Motorbike" ~ 2,
      . == "Public_Transportation" ~ 3,
      . == "Bike" ~ 4,
      . == "Walking" ~ 5,
      TRUE ~ NA_real_
    ))
  )

#       
corr_group <- transformation %>% 
  select(IMT_index, FAVC, CAEC, SMOKE, SCC, CALC, MTRANS,
         family_history_with_overweight, FCVC, NCP, CH2O, FAF, TUE) %>% 
  rename(history = family_history_with_overweight)

#     
corr_group_long <- corr_group %>%
  pivot_longer(cols = -IMT_index,  
               names_to = "Variable", 
               values_to = "Value")
